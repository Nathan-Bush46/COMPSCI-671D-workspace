def euclidean_distance(point1, point2):   
    """takes two np.arrays 
    """
    return np.sqrt(np.sum(np.square(point1 - point2)))

def knn(n,point_to_classify, train_x, df_train_y, threshold):
    k = int(np.sqrt(n)) # get correct k sqrt (n) odd
    if(k % 2 == 0):
        k+=1
    assert k%2 != 0

    distances = []
    for i in range(len(train_x)):
        distances.append(euclidean_distance(point_to_classify, train_x[i]))

    result_df = pd.DataFrame({'distances': distances, 'labels': df_train_y})
    knn = result_df.nsmallest(k,'distances')
    label_counts = knn['labels'].value_counts()
    number_of_1 = label_counts.get(1, 0)
    number_of_0 = label_counts.get(0, 0)
    if(number_of_1/(number_of_1+number_of_0) >= threshold):
        return 1
    return 0

def evaluate(test_predict,test_actual):
    TP = 0
    FP = 0 
    FN = 0 
    TN = 0
    n_test = len(test_actual)
    for i in range(0,n_test):
        if (test_predict[i] == 1 and test_actual[i] == 1):
            TP += 1
        elif(test_predict[i] == 0 and test_actual[i] == 1):
            FN += 1
        elif(test_predict[i] == 0 and test_actual[i] == 0):
            TN += 1
        elif(test_predict[i] == 1 and test_actual[i] == 0):
            FP += 1

    accuracy = (TP +TN)/(TP+FP+FN+TN)
    print(f"Accuracy: {accuracy:.4f}")
    recall = (TP)/(TP+FN)
    print(f"Recall: {recall:.4f}")
    precision = (TP)/(TP+FP)
    print(f"Precision: {precision:.4f}")
    f1 = (2*TP)/(2*TP+FP+FN)
    print(f"F1: {f1:.4f}")

def manhattan_distance(point1, point2):
    return np.sum(np.abs(np.array(point1) - np.array(point2)))

def cosine_similarity(point1, point2):
    dot_product = np.dot(point1, point2)
    norm_a = np.linalg.norm(point1)
    norm_b = np.linalg.norm(point2)
    return 1 - dot_product / (norm_a * norm_b) if norm_a and norm_b else 100

def f_1_score(test_predict,test_actual):
    TP = 0
    FP = 0 
    FN = 0 
    TN = 0
    n_test = len(test_actual)
    for i in range(0,n_test):
        if (test_predict[i] == 1 and test_actual[i] == 1):
            TP += 1
        elif(test_predict[i] == 0 and test_actual[i] == 1):
            FN += 1
        elif(test_predict[i] == 0 and test_actual[i] == 0):
            TN += 1
        elif(test_predict[i] == 1 and test_actual[i] == 0):
            FP += 1
    f1 = (2*TP)/(2*TP+FP+FN)
    return f1

import numpy as np
import matplotlib.pyplot as plt

# Sample data (true labels and predicted scores)
y_true = np.array(test_y_sort)

# Calculate TPR and FPR
thresholds = scores_sort
TPR = []  # True Positive Rate
FPR = []  # False Positive Rate
#y_scores
for threshold in thresholds: # iterate through every threshold and predict based on that value. Get TPR and FPR for each one and graph
    y_pred = (thresholds >= threshold).astype(int)  # Binarize predictions
    TP = np.sum((y_pred == 1) & (y_true == 1))
    FP = np.sum((y_pred == 1) & (y_true == 0))
    FN = np.sum((y_pred == 0) & (y_true == 1))
    TN = np.sum((y_pred == 0) & (y_true == 0))
    
    # cool thing
    TPR.append(TP / (TP + FN) if (TP + FN) > 0 else 0)
    FPR.append(FP / (FP + TN) if (FP + TN) > 0 else 0)

TPR = np.array(TPR)
FPR = np.array(FPR)

# Plot ROC curve with formatting from llm
plt.figure()
plt.plot(FPR, TPR, marker='o', linestyle='-', color='darkorange')
